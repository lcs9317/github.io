@Component
public class UpBitApiClient {
    private final RestTemplate restTemplate = new RestTemplate();

    public CoinPriceResponse getPrice(String symbol) {
        String url = "https://api.upbit.com/v1/ticker?markets=KRW-" + symbol;
        ResponseEntity response = restTemplate.getForEntity(url, String.class);
        JsonNode node = new ObjectMapper().readTree(response.getBody()).get(0);

        return new CoinPriceResponse(
            symbol.toUpperCase(),
            null,
            new BigDecimal(node.get("trade_price").asText()),
            new BigDecimal(node.get("change_rate").asText())
        );
    }
}
